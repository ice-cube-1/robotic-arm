barrels = await scan()
for i in range(len(barrels)):
    await output("starting barrel" +str(i))
    barrels = await pickup(barrels, i)
    await rotate(i*10)
    await move(210, 50)
    barrels = await drop(barrels)
    await move(160, 30)
    await move(160, 200)
    await output(str(i)+" is complete")
